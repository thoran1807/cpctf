### Thought Process

from randcrack import RandCrack
from Crypto.Cipher import AES
from Crypto.Util.Padding import unpad

parts = Path('output.txt').read_text().split()
nums = [int(p) for p in parts if p.isdigit()]
ct_b64 = next(p for p in parts if not p.isdigit())

rc = RandCrack()
for n in nums:

key = hashlib.sha256(str(predicted_password_int).encode()).digest()
plaintext = unpad(AES.new(key, AES.MODE_ECB).decrypt(base64.b64decode(ct_b64)), 16)
